#define REEDPIN   2
#define LEDPIN    4

int value;

void setup() {
  value = 0;
  pinMode(REEDPIN,INPUT);
  pinMode(LEDPIN,OUTPUT);
}

void loop() {
  value = digitalRead(REEDPIN);
  if (value == 0){
    digitalWrite(LEDPIN,HIGH);
    }else{
    digitalWrite(LEDPIN,LOW);
      }
}
